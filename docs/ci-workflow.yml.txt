# CI workflow (parked here for the initial release).
#
# To enable GitHub Actions CI: copy this file to `.github/workflows/ci.yml`. Pushing a file
# under `.github/workflows/` requires a token with the `workflow` OAuth scope — it was parked
# here so the first public release could ship without it. The branch filters are set to
# `master` (this repo's default branch).
#
# It runs the KEYLESS test suite (no QWEN_API_KEY): tests monkeypatch qwen.embed / qwen.chat,
# so recall / forget / bounded-budget behavior is exercised with zero credentials and zero
# network — proving the memory logic independently of the LLM.

name: CI

on:
  push:
    branches: [master]
  pull_request:
    branches: [master]

jobs:
  test:
    runs-on: ubuntu-latest
    strategy:
      fail-fast: false
      matrix:
        # The versions pyproject.toml promises (requires-python = ">=3.9").
        python-version: ["3.9", "3.10", "3.11", "3.12"]

    steps:
      - uses: actions/checkout@v4

      - name: Set up Python ${{ matrix.python-version }}
        uses: actions/setup-python@v5
        with:
          python-version: ${{ matrix.python-version }}

      - name: Install (dev + service extras)
        run: |
          python -m pip install --upgrade pip
          pip install -e ".[dev,service]" httpx

      - name: Run keyless test suite
        env:
          QWEN_API_KEY: ""
        run: pytest -q
